Лабораторная работа: Объекты в JavaScript

Тема: "Объекты в JavaScript" | Выполнил: Шульга Сергей

1. Создание объектов

// Литеральная нотация
let user = {
    name: "Анна",
    age: 25,
    isAdmin: false
};

// Конструктор
let car = new Object();
car.brand = "Toyota";
car.model = "Camry";

2. Доступ к свойствам

let person = {
    name: "Мария",
    "favorite color": "синий"
};

// Разные способы доступа
console.log(person.name); // Через точку
console.log(person["favorite color"]); // Через квадратные скобки

3. Удаление свойства

let book = {
    title: "JavaScript Guide",
    author: "Иванов",
    pages: 300
};

delete book.pages; // Удаление свойства

4. Проверка существования свойства

let animal = {
    type: "cat",
    age: 5
};

// Оператор in
console.log("type" in animal); // true

// Метод hasOwnProperty()
console.log(animal.hasOwnProperty("age")); // true

5. Перебор свойств объекта

let student = {
    name: "Петр",
    grade: 10,
    subject: "Математика"
};

for (let key in student) {
    console.log(key + ": " + student[key]);
}

6. Методы объектов

let calculator = {
    a: 0,
    b: 0,
    setValues: function(x, y) {
        this.a = x;
        this.b = y;
    },
    sum: function() {
        return this.a + this.b;
    }
};
